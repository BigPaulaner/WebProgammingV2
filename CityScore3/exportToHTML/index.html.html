<script>
    const sliders = ['cost', 'air', 'edu', 'safety', 'health'];

    sliders.forEach(id => {
        const slider = document.getElementById(id);
        const valueLabel = document.getElementById(id + '_val');
        slider.addEventListener('input', () => {
            valueLabel.innerText = slider.value;
        });
    });

    function validateWeights() {
        const total = sliders.reduce((sum, id) => sum + parseFloat(document.getElementById(id).value), 0);
        const errorBox = document.getElementById('errorMsg');
        if (Math.abs(total - 1.0) > 0.01) {
            errorBox.classList.remove('d-none');
            return false;
        } else {
            errorBox.classList.add('d-none');
            document.getElementById('loader').style.display = 'flex';
            return true;
        }
    }
</script>
